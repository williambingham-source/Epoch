\section{Theorem Statement}

\subsection*{Definitions}

\begin{definition}
For $x \in \mathbb{R}$, the \emph{fractional part} of $x$ is
$\{x\} = x - \lfloor x \rfloor \in [0,1)$.
The \emph{distance to the nearest integer} is
$\|x\| = \min_{k \in \mathbb{Z}} |x - k| = \min(\{x\}, 1 - \{x\})$.
\end{definition}

\begin{definition}
For $\alpha \in \mathbb{R}$ and $n \in \mathbb{N}$, define the \emph{$n$-point sequence}
\[
  S_n(\alpha) \;=\; \bigl\{\,\{k\alpha\} : 1 \le k \le n\,\bigr\} \;\subset\; [0,1).
\]
Arrange these $n$ distinct points (assuming $\alpha \notin \mathbb{Q}$) in increasing order:
$0 \le x_1 < x_2 < \cdots < x_n < 1$.
The \emph{gaps} are the $n$ half-open arcs
\[
  [x_1, x_2),\; [x_2, x_3),\; \ldots,\; [x_{n-1}, x_n),\;
  [x_n, 1) \cup [0, x_1).
\]
Their lengths are $x_2 - x_1,\; \ldots,\; x_n - x_{n-1},\; 1 - x_n + x_1$
and they sum to $1$.
\end{definition}

\subsection*{The Main Theorem}

\begin{theorem}[Three-Distance Theorem, Sós--Surányi 1958]
  Let $\alpha \in \mathbb{R}$ and $n \in \mathbb{N}$.
  The $n$ gaps determined by $S_n(\alpha)$ on $\mathbb{T} = \mathbb{R}/\mathbb{Z}$
  take \textbf{at most three distinct lengths}.
  Furthermore, if three distinct lengths $s < m < \ell$ occur, then
  \[
    \ell = s + m.
  \]
\end{theorem}

\begin{remark}
  When $\alpha \in \mathbb{Q}$, say $\alpha = p/q$ in lowest terms, the sequence
  $S_n(\alpha)$ is eventually periodic with period $q$, and the gap structure
  is simpler: for $n = q$ there is only \emph{one} distinct gap length, namely $1/q$.
  For $n$ not a multiple of $q$, there are exactly \emph{two} distinct lengths.
\end{remark}

\subsection*{The Gap Lengths Explicitly}

For irrational $\alpha$, the two or three gap lengths can be read off from the
continued fraction $\alpha = [a_0; a_1, a_2, \ldots]$ with convergents $p_k/q_k$.
Given $n$ with $q_m \le n < q_{m+1}$, write
\[
  n = c \cdot q_m + r, \qquad 0 \le r < q_m,
\]
where $c \in \{1, \ldots, a_{m+1}\}$.  Then the gap lengths are
\begin{align*}
  \alpha_m &= \|q_m \alpha\|, \\
  \alpha_{m-1} &= \|q_{m-1} \alpha\|,
\end{align*}
with $\alpha_{m-1} = (c+1)\alpha_m + \alpha_{m+1}$ reflecting the three-length structure
imposed by the Euclidean algorithm step at level $m$.
